openapi: 3.0.3
info:
  title: Highway Digital Twin Service API
  description: >
    Indicative OpenAPI specification for accessing authoritative network,
    scanner condition, forecast condition, and contextual spatial datasets.
  version: "1.0.0"

servers:
  - url: https://api.example.org/v1

paths:

  /authoritative-network/{section_ref}:
    get:
      summary: Get authoritative road network by section reference
      parameters:
        - name: section_ref
          in: path
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Line geometry and attributes for the selected road section
          content:
            application/geo+json:
              schema:
                $ref: "#/components/schemas/GeoJSONFeatureCollection"

  /scanner-condition/historic/{section_ref}:
    get:
      summary: Get historic scanner condition points by section reference
      parameters:
        - name: section_ref
          in: path
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Historic condition point features
          content:
            application/geo+json:
              schema:
                $ref: "#/components/schemas/GeoJSONFeatureCollection"

  /scanner-condition/forecast/{section_ref}:
    get:
      summary: Get forecast scanner condition points by section reference
      parameters:
        - name: section_ref
          in: path
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Forecast condition point features
          content:
            application/geo+json:
              schema:
                $ref: "#/components/schemas/GeoJSONFeatureCollection"

  /geotechnical-inventory/{section_ref}:
    get:
      summary: Get geotechnical inventory polygons intersecting a road section
      parameters:
        - name: section_ref
          in: path
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Intersecting geotechnical polygon features
          content:
            application/geo+json:
              schema:
                $ref: "#/components/schemas/GeoJSONFeatureCollection"

  /environmental-context/{section_ref}:
    get:
      summary: Get environmental context polygons intersecting a road section
      parameters:
        - name: section_ref
          in: path
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Intersecting environmental context polygon features
          content:
            application/geo+json:
              schema:
                $ref: "#/components/schemas/GeoJSONFeatureCollection"


  /digital-twin/section/{section_ref}:
    get:
      summary: Get full multi-layer digital twin view for a road section
      description: >
        Returns authoritative network geometry, historic condition, forecast
        condition, geotechnical inventory, and environmental context as a single
        structured GeoJSON object.
      parameters:
        - name: section_ref
          in: path
          required: true
          schema:
            type: string
      responses:
        "200":
          description: Full digital twin GeoJSON payload
          content:
            application/json:
              schema:
                $ref: "#/components/schemas/DigitalTwinResponse"


components:

  schemas:

    GeoJSONFeatureCollection:
      type: object
      properties:
        type:
          type: string
          example: FeatureCollection
        features:
          type: array
          items:
            type: object

    DigitalTwinResponse:
      type: object
      properties:
        authoritative network:
          $ref: "#/components/schemas/GeoJSONFeatureCollection"
        historic condition:
          $ref: "#/components/schemas/GeoJSONFeatureCollection"
        forecast condition:
          $ref: "#/components/schemas/GeoJSONFeatureCollection"
        geotechnical inventory:
          $ref: "#/components/schemas/GeoJSONFeatureCollection"
        context:
          $ref: "#/components/schemas/GeoJSONFeatureCollection"
